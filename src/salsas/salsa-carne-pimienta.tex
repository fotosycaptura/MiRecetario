\begin{recipe}[source={Youtube},
	preparationtime={\unit[8-15]{Minutos}}
	]{Salsa de carne a la pimienta}
	\introduction{
		Esta preparación de salsa para asado la vi en un short de youtube en sus días, me pareció interesante y después de probarla me gustó.
	}
	\ingredients{
		1 & cucharada de pimienta mitad molida, mitad entera \\
		200 CC & de crema de leche \\
		& Ajo a gusto \\
		& Sal a gusto \\
		1& caldo de carne concentrado \\
		& Aceite de Oliva o un cubo de mantequilla
	}
	\preparation{
		\begin{enumerate}
			\item Agregar la mantequilla o el aceite en la sartén y calentar.
			\item Una vez caliente o derretida en el caso de la mantequilla, dejar fuego suave y agregar la pimienta, y comenzar a revolver por 5 minutos a fuego suave.
			\item Transcurrido ese tiempo, agregar la crema de leche y una pisca de sal y revolver constantemente, luego agregar el caldo de carne y revolver por otros 10 minutos aproximadamente. 
			\item Dejar enfriar un poco y servir encima de alguna chuleta o carne que se haya preparado ya sea al horno o frita.
		\end{enumerate}
	}
\end{recipe}
